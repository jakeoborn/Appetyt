  async signInWithApple(){
    const sb = getSB();
    if(!sb){ this.showToast('Sign-in unavailable -- try again later'); return; }
    try{
      // Native iOS: use Capacitor plugin so Apple's system sheet appears (no webview redirect)
      if (window.Capacitor && window.Capacitor.isNativePlatform()) {
        const SignInWithApple = window.Capacitor.Plugins.SignInWithApple;
        if (SignInWithApple) {
          const result = await SignInWithApple.authorize({
            clientId: 'com.dimhour.app',
            redirectURI: 'https://nsmwgokjjnpwwvearwxf.supabase.co/auth/v1/callback',
            scopes: 'email name'
          });
          const idToken = result && result.response && result.response.identityToken;
          if (!idToken) throw new Error('No identity token from Apple');
          const {error: idErr} = await sb.auth.signInWithIdToken({
            provider: 'apple',
            token: idToken
          });
          if (idErr) throw idErr;
          return;
        }
      }
      // Web fallback: redirect-based OAuth
      const {data, error} = await sb.auth.signInWithOAuth({
        provider: 'apple',
        options: { redirectTo: 'https://dimhour.com' }
      });
      if(error){ console.error('[Apple OAuth]', error); throw error; }
    } catch(e){
      console.error('[Apple OAuth]', e);
      const errEl = document.getElementById('auth-err');
      errEl.textContent = 'Apple sign-in failed: ' + (e.message||'Try email instead');
      errEl.style.display = 'block';
    }
  },
